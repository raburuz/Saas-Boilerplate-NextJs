Create a Textarea, you can use your own custom component.

``` js

/* BOILERPLATE */
import { Form, FormField, FormLabel, FormMessage, Textarea } from '@/boilerplate/form'

/* COMPONENT */
export const Component = () => {
  
  return (
    <Form>
      <FormField>
        <FormLabel> Description: </FormLabel>
        <Textarea name="description"/>
        <FormMessage message={"Description is required"} />
        {/*Code*/}
      </FormField>

      <button type="submit">Send</button>
      
    </Form>
  )

}

```
