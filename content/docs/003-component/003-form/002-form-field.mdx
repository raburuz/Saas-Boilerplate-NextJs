Create a form field, you can use your own custom component.

``` js

/* BOILERPLATE */
import { Form, FormField } from '@/boilerplate/form'

/* COMPONENT */
export const Component = () => {
  
  return (
    <Form>
      <FormField>
        {/*Code*/}
      </FormField>

      <button type="submit">Send</button>
      
    </Form>
  )

}

```
