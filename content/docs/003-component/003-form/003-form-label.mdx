Create a form label, you can use your own custom component.

``` js

/* BOILERPLATE */
import { Form, FormField, FormLabel } from '@/boilerplate/form'

/* COMPONENT */
export const Component = () => {
  
  return (
    <Form>
      <FormField>
        <FormLabel> Name: </FormLabel>
        {/*Code*/}
      </FormField>

      <button type="submit">Send</button>
      
    </Form>
  )

}

```
