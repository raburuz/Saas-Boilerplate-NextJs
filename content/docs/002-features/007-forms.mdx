Founder use [Zod](https://zod.dev/) and [React Form Hook](https://react-hook-form.com/) to made easy work with form validation

``` js

  "use client"

  /* LIBRARIES */
  import { useForm } from "react-hook-form";
  import { zodResolver } from '@hookform/resolvers/zod';
  import { z } from "zod";

  /* BOILERPLATE */
  import { InferSchemaValues } from "@/lib/zod";

  /* ZOD SCHEMA */
  const schema = z.object({
    name: 
      z.string()
      .trim()
      .min( 1 , "Name is required")
  })

  type ISchema = InferSchemaValues<typeof schema>

  /* COMPONENT */
  export const Form = () => {

    //hook
    const form = useForm<ISchema>({
      resolver: zodResolver(schema),
      mode:'onBlur',
    });

    //function
    const onSubmit = ( formData: ISchema ) => { 
      // Code
    };

    //jsx
    return (
      <>

        <form onSubmit={form.handleSubmit(onSubmit)}>
          <input type="text" {...form.register("name")}/>
          <button type="submit"> ok </button>
        </form>

      </>
    )
  }

```
